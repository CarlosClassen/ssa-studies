workspace {

  model {
    student = person "Student" "A student at the university."
    professor = person "Professor" "A professor at the university."

    projectSystem = softwareSystem "Project System" "Allows professors to create and manage student teams and projects." {
        projectWebApp = container "Project Web Application" "Enables interaction with project-related data." "Web Application" {
            teamManagementComponent = component "Team Management Component" "Allows professors to create teams."
            projectManagementComponent = component "Project Management Component" "Allows professors to create projects."
            assignmentComponent = component "Assignment Component" "Handles assigning projects to teams."
            submissionComponent = component "Submission Component" "Allows students to submit their projects."
            
            professor -> teamManagementComponent "Requests to create team"
            professor -> projectManagementComponent "Requests to create project"
            professor -> assignmentComponent "Requests to assign project to team"
            student -> submissionComponent "Submits project"
        }

        db = container "Database" "Stores all the project-related data." "Relational Database Schema" {
            teamsTable = component "Teams Table" "Stores information about teams."
            projectsTable = component "Projects Table" "Stores information about projects."
            submissionsTable = component "Submissions Table" "Stores information about project submissions."
        }

        projectWebApp -> db "Reads from and writes to"
        teamManagementComponent -> teamsTable "Validates team information and writes to Teams Table"
        projectManagementComponent -> projectsTable "Validates project information and writes to Projects Table"
        assignmentComponent -> teamsTable "Reads from Teams Table"
        assignmentComponent -> projectsTable "Writes to Projects Table"
        submissionComponent -> submissionsTable "Validates submissions and writes to Submissions Table"
    }

    student -> projectWebApp "Uses"
    professor -> projectWebApp "Uses"
  }

  views {
    systemContext projectSystem {
      include student
      include professor
      autolayout lr
    }

    container projectSystem {
      include *
      autolayout lr
    }

    component projectWebApp {
      include *
      autolayout lr
    }

    styles {
      element "Software System" {
        background #1168bd
        color #ffffff
      }
      element "Container" {
        background #438dd5
        color #ffffff
      }
      element "Component" {
        background #85bbf0
        color #000000
      }
      element "Person" {
        shape person
        background #08427b
        color #ffffff
      }
    }
  }
}
